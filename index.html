        function minNum(...num) {
            num.sort();
            for(let i = 0; i < num.length; i++)
            {
                for(let x = 0; x < num.length; x++)
                {
                    if(num[i] < num[i + 1])
                    {
                        return num[i];
                    }
                }
            }
        }
